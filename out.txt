#region 进阶功能：实时更新窗⼝信息
public override void _Process(double delta)
{
    // 计时器累加
    _updateTimer += (float)delta;
    // 达到间隔，重新枚举窗⼝
    if (_updateTimer >= UpdateInterval)
    {
        _updateTimer = 0f;
        // 保存当前选中窗⼝的句柄，刷新后重新选中
        IntPtr selectedHwnd = SelectedWindow?.Hwnd ?? IntPtr.Zero;
        // 重新枚举外部窗⼝
        EnumerateExternalWindows();
        // 刷新后重新选中原窗⼝（保持绘制不变）
        if (selectedHwnd != IntPtr.Zero)
        {
            var newSelectedWindow = FindWindowByHwnd(selectedHwnd);
            if (newSelectedWindow != null)
            {
                SelectWindowAndRedraw(newSelectedWindow);
            }
            else
            {
                SelectedWindow = null;
                QueueRedraw(); // 窗⼝已关闭，清空绘制
            }
        }
    }
}
#endregion
